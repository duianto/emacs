* 始めに
  辞書を引くための機能.
* 読み込み
  #+BEGIN_SRC emacs-lisp
    (autoload 'lookup "lookup" nil t)
    (autoload 'lookup-region "lookup" nil t)
    (autoload 'lookup-pattern "lookup" nil t)
  #+END_SRC
* キーバインドの設定
  =\C-cw= で辞書を引く
  #+BEGIN_SRC emacs-lisp
    (global-set-key "\C-cw" 'lookup-pattern)
    (global-set-key "\C-cW" 'lookup-word)
#+END_SRC
