#+BEGIN_SRC emacs-lisp
  (my:not-locate-library window-layout "site-lisp/emacs-window-layout")
  (my:not-locate-library e2wm-config "site-lisp/emacs-window-manager")
  (require 'e2wm-config)
  (global-set-key (kbd "M-+") 'e2wm:start-management)
#+END_SRC
